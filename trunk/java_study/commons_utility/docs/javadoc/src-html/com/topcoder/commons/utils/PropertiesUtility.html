009    /**
010     * This is a utility class that provides static methods for retrieving
011     * properties of different types (String, Integer, Long, Double, Date, Class)
012     * from Properties instance. This utility throws en exception specified by the
013     * caller if required property is missing or cannot be parsed properly.
014     * getSubConfiguration() method allows to extract inner configuration from
015     * Properties instance (when "childConfigName.childPropertyName" format is used
016     * for property keys).
017     * 
018     * Thread Safety: This class is immutable and thread safe when properties
019     * parameters passed to it are used by the caller in thread safe manner.
020     */
021    public class PropertiesUtility {
022        /**
023         * Empty private constructor.
024         * 
025         * Implementation Notes: Do nothing.
026         */
027        private PropertiesUtility() {
028        }

030        /**
031         * Retrieves the string property from the given Properties instance.
032         * 
033         * Generic Parameters: T - the type of the exception to be thrown if some
034         * error occurred
035         * 
036         * Parameters: properties - the properties container key - the key of the
037         * property to be retrieved required - true if property is required, false
038         * otherwise (if property is required, but missing, an exception is thrown)
039         * exceptionClass - the type of the exception to be thrown if some error
040         * occurs
041         * 
042         * Returns: the retrieved string property value (null if property is
043         * optional and missing)
044         * 
045         * Throws: T if the property is required, but missing
046         * 
047         * Implementation Notes: 1. String value = properties.getProperty(key); 2.
048         * If value == null then 2.1. If required then 2.1.1. Throw
049         * ExceptionHelper.constructException(exceptionClass, getPropertyTitle(key)
050         * + " is required"); 2.2. Return null. 3. Return value.
051         * 
052         * @throws T
053         *             if the property is required, but missing
054         * @param exceptionClass
055         *            the type of the exception to be thrown if some error occurs
056         * @param properties
057         *            the properties container
058         * @param required
059         *            true if property is required, false otherwise (if property is
060         *            required, but missing, an exception is thrown)
061         * @param key
062         *            the key of the property to be retrieved
063         * @return the retrieved string property value (null if property is optional
064         *         and missing)
065         */
066        public static <T extends Throwable> String getStringProperty(
067                Properties properties, String key, boolean required,
068                Class<T> exceptionClass) throws T {
069            String value = properties.getProperty(key);
070            if (value == null && required) {
071                throw ExceptionHelper.constructException(exceptionClass, getPropertyTitle(key) + " is required");
072            }
073            return value;
074        }

076        /**
077         * Retrieves the delimited strings property from the given Properties
078         * instance.
079         * 
080         * Generic Parameters: T - the type of the exception to be thrown if some
081         * error occurred
082         * 
083         * Parameters: properties - the properties container key - the key of the
084         * property to be retrieved delimiter - the delimiter regular expression
085         * pattern required - true if property is required, false otherwise (if
086         * property is required, but missing, an exception is thrown) exceptionClass
087         * - the type of the exception to be thrown if some error occurs
088         * 
089         * Returns: the retrieved strings values (null if property is optional and
090         * missing)
091         * 
092         * Throws: T if the property is required, but missing
093         * 
094         * Implementation Notes: 1. String value = properties.getProperty(key); 2.
095         * If value == null then 2.1. If required then 2.1.1. Throw
096         * ExceptionHelper.constructException(exceptionClass, getPropertyTitle(key)
097         * + " is required"); 2.2. Return null. 3. Return value.split(delimiter,
098         * -1).
099         * 
100         * @throws T
101         *             if the property is required, but missing
102         * @param delimiter
103         *            the delimiter regular expression pattern
104         * @param exceptionClass
105         *            the type of the exception to be thrown if some error occurs
106         * @param properties
107         *            the properties container
108         * @param required
109         *            true if property is required, false otherwise (if property is
110         *            required, but missing, an exception is thrown)
111         * @param key
112         *            the key of the property to be retrieved
113         * @return the retrieved strings values (null if property is optional and
114         *         missing)
115         */
116        public static <T extends Throwable> String[] getStringsProperty(
117                Properties properties, String key, String delimiter,
118                boolean required, Class<T> exceptionClass) throws T {
119            String value = properties.getProperty(key);
120            if (value == null) {
121                if (required) {
122                    throw ExceptionHelper.constructException(exceptionClass, getPropertyTitle(key) + " is required");
123                }else{
124                    return null;
125                }   
126            }
127            
128            return value.split(delimiter);
129        }

135         * error occurred

138         * property to be retrieved required - true if property is required, false
139         * otherwise (if property is required, but missing, an exception is thrown)
140         * exceptionClass - the type of the exception to be thrown if some error
141         * occurs
142         * 
143         * Returns: the retrieved integer property value (null if property is
144         * optional and missing)
145         * 
146         * Throws: T if the property value has invalid format or is required, but
147         * missing
148         * 
149         * Implementation Notes: 1. String value = properties.getProperty(key); 2.
150         * If value == null then 2.1. If required then 2.1.1. Throw
151         * ExceptionHelper.constructException(exceptionClass, getPropertyTitle(key)
152         * + " is required"); 2.2. Return null. 3. Integer result =
153         * Integer.valueOf(value); 4. If ex:NumberFormatException is thrown then
154         * 4.1. Throw ExceptionHelper.constructException(exceptionClass,
155         * getPropertyTitle(key) + " should be a valid integer", ex); 5. Return
156         * result.
157         * 
158         * @throws T
159         *             if the property value has invalid format or is required, but
160         *             missing
161         * @param exceptionClass
162         *            the type of the exception to be thrown if some error occurs
163         * @param properties
164         *            the properties container
165         * @param required
166         *            true if property is required, false otherwise (if property is

169         *            the key of the property to be retrieved
170         * @return the retrieved integer property value (null if property is
171         *         optional and missing)
172         */
173        public static <T extends Throwable> Integer getIntegerProperty(
174                Properties properties, String key, boolean required,
175                Class<T> exceptionClass) throws T {
176            String value = properties.getProperty(key);
177            if (value == null) {
178                if (required) {
179                    throw ExceptionHelper.constructException(exceptionClass, getPropertyTitle(key) + " is required");
180                }else{
181                    return null;
182                }   
183            }
184            
185            try {
186                return Integer.parseInt(value);
187            } catch (NumberFormatException e) {
188                throw ExceptionHelper.constructException(exceptionClass, getPropertyTitle(key) + " should be a valid integer", e);
189            }
190        }

444        /**
445         * Retrieves the inner configuration from the configuration stored in
446         * Properties container.
447         * 
448         * Parameters: properties - the properties with the main configuration
449         * configName - the name of the inner configuration
450         * 
451         * Returns: the Properties container with the extracted inner configuration
452         * (not null)
453         * 
454         * Implementation Notes: 1. String prefix = configName + "."; 2. Create
455         * Properties instance for inner configuration: result = new Properties();
456         * 3. For each (String key; String value) from properties do: 3.1. If
457         * key.startsWith(prefix) then 3.1.1. Remove the prefix from the key: String
458         * newKey = key.substring(prefix.length()); 3.1.2. Put key/value pair to the
459         * inner configuration: result.put(newKey, value); 4. Return result.
460         * 
461         * @param configName
462         *            the name of the inner configuration
463         * @param properties
464         *            the properties with the main configuration
465         * @return the Properties container with the extracted inner configuration
466         *         (not null)
467         */
468        public static Properties getSubConfiguration(Properties properties,
469                String configName) {
470            String prefix = configName + ".";
471            Set<Object> keys = properties.keySet();
472            Properties result = new Properties();
473            for (Object keyO : keys) {
474                String key = keyO.toString();
475                if (key.startsWith(prefix)) {
476                    String value = properties.getProperty(key);
477                    String newKey = key.substring(prefix.length());
478                    result.put(newKey, value);
479                }
480            }
481            
482            return result;
483        }
