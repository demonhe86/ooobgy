028        /**
029         * Executes the SQL query and parses the query result from the ResultSet.
030         * 
031         * Generic Parameters: T - the type of the exception to be thrown if some
032         * error occurred
033         * 
034         * Parameters: connection - the database connection to be used queryString -
035         * the query string argumentTypes - the argument types (see java.sql.Types)
036         * queryArgs - the query argument values columnTypes - the result column
037         * types (can be one of Date.class, Double.class, Float.class, Long.class,
038         * Integer.class, String.class, Boolean.class, Object.class)
039         * persistenceExceptionClass - the class of persistence exception to be
040         * thrown if some error occurs
041         * 
042         * Returns: the SQL query result; outer array represents rows; inner arrays
043         * represent columns (not null)
044         * 
045         * Throws: T if some error occurred when executing query or parsing result
046         * 
047         * Implementation Notes: 1. Prepare statement: PreparedStatement
048         * preparedStatement = connection.prepareStatement(queryString); 2. For i =
049         * 0 .. argumentTypes.length - 1 do: 2.1. Set parameter to the prepared
050         * statement: preparedStatement.setObject(i + 1, queryArgs[i],
051         * argumentTypes[i]); 3. Execute SQL query: ResultSet resultSet =
052         * preparedStatement.executeQuery(); 4. Create a list for retrieved rows:
053         * result = new ArrayList<Object>(); 5. Get metadata from the result set:
054         * ResultSetMetaData metaData = resultSet.getMetaData(); 6. Get the number
055         * of columns in the result set: int columnCount =
056         * metaData.getColumnCount(); 7. If columnTypes.length != columnCount then
057         * 7.1. Throw ExceptionHelper.constructException("The column types length ["
058         * + columnTypes.length + "] does not match the result set column count[" +
059         * columnCount + "]."); 8. While resultSet.next() do: 8.1. Object[] rowData
060         * = new Object[columnCount]; 8.2. For i = 0 .. columnCount - 1 do: 8.2.1.
061         * Object cellValue; 8.2.2. If columnTypes[i] == Date.class then 8.2.2.1.
062         * Timestamp timestamp = resultSet.getTimestamp(i + 1); 8.2.2.2. cellValue =
063         * timestamp == null ? null : new Date(timestamp.getTime()); 8.2.3. Else if
064         * columnTypes[i] == Double.class then 8.2.3.1. cellValue =
065         * resultSet.getDouble(i + 1); 8.2.3.2. If resultSet.wasNull() then
066         * cellValue = null; 8.2.4. Else if columnTypes[i] == Float.class then
067         * 8.2.4.1. cellValue = resultSet.getFloat(i + 1); 8.2.4.2. If
068         * resultSet.wasNull() then cellValue = null; 8.2.5. Else if columnTypes[i]
069         * == Long.class then 8.2.5.1. cellValue = resultSet.getLong(i + 1);
070         * 8.2.5.2. If resultSet.wasNull() then cellValue = null; 8.2.6. Else if
071         * columnTypes[i] == Integer.class then 8.2.6.1. cellValue =
072         * resultSet.getInt(i + 1); 8.2.6.2. If resultSet.wasNull() then cellValue =
073         * null; 8.2.7. Else if columnTypes[i] == String.class then 8.2.7.1.
074         * cellValue = resultSet.getString(i + 1); 8.2.8. Else if columnTypes[i] ==
075         * Boolean.class then 8.2.8.1. cellValue = resultSet.getBoolean(i + 1);
076         * 8.2.8.2. If resultSet.wasNull() then cellValue = null; 8.2.9. Else if
077         * columnTypes[i] == Object.class then 8.2.9.1. cellValue =
078         * resultSet.getObject(i + 1); 8.2.10. Else Throw
079         * ExceptionHelper.constructException(exceptionClass,
080         * "Unsupported column type is used: " + columnTypes[i].getName()); 8.2.11.
081         * rowData[i] = cellValue; 8.3. Add row to the result list:
082         * result.add(rowData); 9. If ex:SQLException was thrown then 9.1. Throw
083         * ExceptionHelper.constructException(exceptionClass,
084         * "Error occurred while executing query [" + queryString +
085         * "] using the query arguments " + Arrays.asList(queryArgs).toString() +
086         * ".", ex). 10. resultSet.close(); // do in "finally" block, ignore
087         * exceptions 11. preparedStatement.close(); // do in "finally" block,
088         * ignore exceptions 12. Return result.toArray(new Object[][]{}).
089         * 
090         * @throws T
091         *             if some error occurred when executing query or parsing result
092         * @param persistenceExceptionClass
093         *            the class of persistence exception to be thrown if some error
094         *            occurs
095         * @param queryString
096         *            the query string
097         * @param connection
098         *            the database connection to be used
099         * @param columnTypes
100         *            the result column types (can be one of Date.class,
101         *            Double.class, Float.class, Long.class, Integer.class,
102         *            String.class, Boolean.class, Object.class)
103         * @param argumentTypes
104         *            the argument types (see java.sql.Types)
105         * @param queryArgs
106         *            the query argument values
107         * @return the SQL query result; outer array represents rows; inner arrays
108         *         represent columns (not null)
109         */
110        public static <T extends Throwable> Object[][] executeQuery(
111                Connection connection, String queryString, int[] argumentTypes,
112                Object[] queryArgs, Class<?>[] columnTypes,
113                Class<T> persistenceExceptionClass) {
114            return null;
115        }
