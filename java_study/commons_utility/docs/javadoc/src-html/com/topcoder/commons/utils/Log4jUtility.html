001    package com.topcoder.commons.utils;
002    
003    import java.util.Date;
004    
005    import org.apache.log4j.Level;
006    import org.apache.log4j.Logger;
007    
008    /**
009     * This is a utility class that provides static methods for logging method
010     * entrance, method exit and exception using Log4j library. It allows to
011     * optionally log method input/output parameters and method execution time.
012     * Exceptions are logged together with stack traces. Default priority for method
013     * entrance/exit message is DEBUG, for exceptions - ERROR; but both can be
014     * overridden.
015     * 
016     * Thread Safety: This class is immutable and thread safe when array parameters
017     * passed to it are used by the caller in thread safe manner.
018     */
019    public class Log4jUtility {
020        /**
021         * Empty private constructor.
022         * 
023         * Implementation Notes: Do nothing.
024         */
025        private Log4jUtility() {
026        }
027    
028        /**
029         * Logs the method entrance together with input parameters (if present).
030         * It's assumed that paramNames and paramValues contain the same number of
031         * elements. When this method is used, DEBUG priority is used.
032         * 
033         * Parameters: logger - the logger to be used (null if logging is not
034         * required to be performed) signature - the signature that uniquely
035         * identifies the method (e.g. className#methodName) paramNames - the names
036         * of input parameters (null of method doesn't accept any parameters)
037         * paramValues - the values of input parameters
038         * 
039         * Implementation Notes: 1. logEntrance(logger, signature, paramNames,
040         * paramValues, Level.DEBUG);
041         * 
042         * @param logger
043         *            the logger to be used (null if logging is not required to be
044         *            performed)
045         * @param paramValues
046         *            the values of input parameters
047         * @param signature
048         *            the signature that uniquely identifies the method (e.g.
049         *            className#methodName)
050         * @param paramNames
051         *            the names of input parameters (null of method doesn't accept
052         *            any parameters)
053         */
054        public static void logEntrance(Logger logger, String signature,
055                String[] paramNames, Object[] paramValues) {
056            logEntrance(logger, signature, paramNames, paramValues, Level.DEBUG);
057        }
058    
059        /**
060         * Logs the method entrance together with input parameters (if present) and
061         * timestamp (optionally). It's assumed that paramNames and paramValues
062         * contain the same number of elements.
063         * 
064         * Parameters: logger - the logger to be used (null if logging is not
065         * required to be performed) signature - the signature that uniquely
066         * identifies the method (e.g. className#methodName) paramNames - the names
067         * of input parameters (null of method doesn't accept any parameters)
068         * paramValues - the values of input parameters priority - the logging
069         * priority to be used
070         * 
071         * Implementation Notes: 1. If logger == null then return; 2.
072         * logger.log(priority,
073         * LoggingUtilityHelper.getMethodEntranceMessage(signature)); 3. If
074         * paramNames != null then 3.1. logger.log(priority,
075         * LoggingUtilityHelper.getInputParametersMessage(paramNames, paramValues));
076         * 
077         * @param level
078         *            the logging level to be used
079         * @param logger
080         *            the logger to be used (null if logging is not required to be
081         *            performed)
082         * @param paramValues
083         *            the values of input parameters
084         * @param signature
085         *            the signature that uniquely identifies the method (e.g.
086         *            className#methodName)
087         * @param paramNames
088         *            the names of input parameters (null of method doesn't accept
089         *            any parameters)
090         */
091        public static void logEntrance(Logger logger, String signature,
092                String[] paramNames, Object[] paramValues, Level level) {
093            if (logger == null) {
094                return;
095            }
096            logger.log(level, LoggingUtilityHelper.getInputParametersMessage(paramNames, paramValues));
097        }
098    
099        /**
100         * Logs the method exit together with the returned value (if present). When
101         * this method is used, method execution time is not logged and DEBUG
102         * priority is used.
103         * 
104         * Parameters: logger - the logger to be used (null if logging is not
105         * required to be performed) signature - the signature that uniquely
106         * identifies the method (e.g. className#methodName) value - the value
107         * returned from the method (should contain 1 element with the returned
108         * value, or should be null if the method returns void)
109         * 
110         * Implementation Notes: 1. logExit(logger, signature, value, null);
111         * 
112         * @param value
113         *            the value returned from the method (should contain 1 element
114         *            with the returned value, or should be null if the method
115         *            returns void)
116         * @param logger
117         *            the logger to be used (null if logging is not required to be
118         *            performed)
119         * @param signature
120         *            the signature that uniquely identifies the method (e.g.
121         *            className#methodName)
122         */
123        public static void logExit(Logger logger, String signature, Object[] value) {
124            logExit(logger, signature, value, null);
125        }
126    
127        /**
128         * Logs the method exit together with the returned value (if present) and
129         * method execution time. When this method is used, DEBUG priority is used.
130         * 
131         * Parameters: logger - the logger to be used (null if logging is not
132         * required to be performed) signature - the signature that uniquely
133         * identifies the method (e.g. className#methodName) value - the value
134         * returned from the method (should contain 1 element with the returned
135         * value, or should be null if the method returns void) entranceTimestamp -
136         * the method entrance timestamp (null if not available), is used for
137         * calculating method execution time
138         * 
139         * Implementation Notes: 1. logExit(logger, signature, value,
140         * entranceTimestamp, Level.DEBUG);
141         * 
142         * @param value
143         *            the value returned from the method (should contain 1 element
144         *            with the returned value, or should be null if the method
145         *            returns void)
146         * @param logger
147         *            the logger to be used (null if logging is not required to be
148         *            performed)
149         * @param entranceTimestamp
150         *            the method entrance timestamp (null if not available), is used
151         *            for calculating method execution time
152         * @param signature
153         *            the signature that uniquely identifies the method (e.g.
154         *            className#methodName)
155         */
156        public static void logExit(Logger logger, String signature, Object[] value,
157                Date entranceTimestamp) {
158            logExit(logger, signature, value, entranceTimestamp, Level.DEBUG);
159        }
160    
161        /**
162         * Logs the method exit together with the returned value (if present) and
163         * method execution time.
164         * 
165         * Parameters: logger - the logger to be used (null if logging is not
166         * required to be performed) signature - the signature that uniquely
167         * identifies the method (e.g. className#methodName) value - the value
168         * returned from the method (should contain 1 element with the returned
169         * value, or should be null if the method returns void) entranceTimestamp -
170         * the method entrance timestamp (null if not available), is used for
171         * calculating method execution time priority - the logging priority to be
172         * used
173         * 
174         * Implementation Notes: 1. If logger == null then return; 2.
175         * logger.log(priority, LoggingUtilityHelper.getMethodExitMessage(signature,
176         * entranceTimestamp)); 3. If value != null then 3.1. logger.log(priority,
177         * LoggingUtilityHelper.getOutputValueMessage(value[0]));
178         * 
179         * @param level
180         *            the logging level to be used
181         * @param value
182         *            the value returned from the method (should contain 1 element
183         *            with the returned value, or should be null if the method
184         *            returns void)
185         * @param logger
186         *            the logger to be used (null if logging is not required to be
187         *            performed)
188         * @param entranceTimestamp
189         *            the method entrance timestamp (null if not available), is used
190         *            for calculating method execution time
191         * @param signature
192         *            the signature that uniquely identifies the method (e.g.
193         *            className#methodName)
194         */
195        public static void logExit(Logger logger, String signature, Object[] value,
196                Date entranceTimestamp, Level level) {
197            if (logger == null) {
198                return;
199            }
200            logger.log(level, LoggingUtilityHelper.getMethodExitMessage(signature, entranceTimestamp));
201            if (value != null) {
202                logger.log(level, LoggingUtilityHelper.getOutputValueMessage(value[0]));
203            }
204        }
205    
206        /**
207         * Logs the given exception. When this method is used, ERROR priority is
208         * used.
209         * 
210         * Generic Parameters: T - the type of the exception to be logged and
211         * returned
212         * 
213         * Parameters: logger - the logger to be used (null if logging is not
214         * required to be performed) signature - the signature that uniquely
215         * identifies the method (e.g. className#methodName) where the exception is
216         * logged exception - the exception to be logged (assumed to be not null)
217         * 
218         * Returns: the logged exception
219         * 
220         * Implementation Notes: 1. Return logException(logger, signature,
221         * exception, Level.ERROR).
222         * 
223         * @param exception
224         *            the exception to be logged (assumed to be not null)
225         * @param logger
226         *            the logger to be used (null if logging is not required to be
227         *            performed)
228         * @param signature
229         *            the signature that uniquely identifies the method (e.g.
230         *            className#methodName) where the exception is logged
231         * @return the logged exception
232         */
233        public static <T extends Throwable> T logException(Logger logger,
234                String signature, T exception) {
235            return logException(logger, signature, exception, Level.ERROR);
236        }
237    
238        /**
239         * Logs the given exception using the specified priority.
240         * 
241         * Generic Parameters: T - the type of the exception to be logged and
242         * returned
243         * 
244         * Parameters: logger - the logger to be used (null if logging is not
245         * required to be performed) signature - the signature that uniquely
246         * identifies the method (e.g. className#methodName) where the exception is
247         * logged exception - the exception to be logged (assumed to be not null)
248         * priority - the logging priority to be used
249         * 
250         * Returns: the logged exception
251         * 
252         * Implementation Notes: 1. If logger == null then return; 2.
253         * logger.log(priority, LoggingUtilityHelper.getExceptionMessage(signature,
254         * exception)); 1. Return exception.
255         * 
256         * @param level
257         *            the logging level to be used
258         * @param exception
259         *            the exception to be logged (assumed to be not null)
260         * @param logger
261         *            the logger to be used (null if logging is not required to be
262         *            performed)
263         * @param signature
264         *            the signature that uniquely identifies the method (e.g.
265         *            className#methodName) where the exception is logged
266         * @return the logged exception
267         */
268        public static <T extends Throwable> T logException(Logger logger,
269                String signature, T exception, Level level) {
270            if (logger == null) {
271                return exception;
272            }
273            logger.log(level, LoggingUtilityHelper.getExceptionMessage(signature, exception));
274            return exception;
275        }
276    }
