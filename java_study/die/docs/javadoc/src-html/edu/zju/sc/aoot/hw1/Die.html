001    package edu.zju.sc.aoot.hw1;
002    
003    import java.util.Random;
004    
005    /**
006     * A n-sided die.<p/>
007     * You can create a default 6-sided die with {@link #Die()}.<br/>
008     * You can also create a die of specified side count with {@link #Die(int)}.<br/>
009     * Get the count of die side with {@link #numOfSides()}<br/>
010     * {@link #roll()} count roll the die and return the top current.<br/>
011     * {@link #topFace()} return the current top without rolling the die
012     * 
013     * @author Frogcherry ÖÜÏþÁú frogcherry@gmail.com
014     * @since  2011-9-13
015     * @version 1.0.0
016     */
017    public class Die {
018            /** side count of the die */
019            private final int sideCnt;
020            /** current top of the die */
021            private int top;
022            /** a simple random factory used {@link Random} */
023            private Random randFatory = new Random();
024            
025            /**
026             * Constructor without arguments default appoints the die with 6 sides.
027             */
028            public Die() {
029                    this.sideCnt = 6;
030                    roll();
031            }
032            
033            /**
034             * This Constructor create a die with specified side count.
035             * @param sideCnt
036             */
037            public Die(int sideCnt) {
038                    if (sideCnt < 1) {
039                            throw new IllegalArgumentException();
040                    }
041                    this.sideCnt = sideCnt;
042                    roll();
043            }
044            
045            /**
046             * Get the number of side of this die.
047             * @return {@link #sideCnt}  (side count of the die)
048             */
049            public int numOfSides(){
050                    return sideCnt;
051            }
052            
053            /**
054             * Roll the die and return the current top.
055             * @return {@link #top}  (the current top)
056             */
057            public int roll(){
058                    this.top = randFatory.nextInt(sideCnt) + 1;
059                    
060                    return this.top;
061            }
062            
063            /**
064             * Return the current top without rolling the die.
065             * @return {@link #top}  (the current top)
066             */
067            public int topFace(){           
068                    return this.top;
069            }
070    }
